def check_prime(num):
    for i in range(2, num):
        if num % i == 0:
            return False
        else:
            continue
    return True
    
def main():
    a = 13
    b = 15

    if check_prime(a):
        print(str(a) + ' .')
    else:
        print(str(a) + '  .')
        
    if check_prime(b):
        print(str(b) + ' .')
    else:
        print(str(b) + '  .')

main()

def add_comma(val):
    result, comma = "", ""
    cnt = 0

    number = str(val)
    number = number[len(number)::-1]

    for i in range(len(number)):
        cnt += 1
        if cnt % 3 == 0:
            result += number[i] + ','
        else :
            result += number[i]

    return result[::-1] + comma

def main():
    comma_added_1234 = add_comma(1234)
    comma_added_12345678 = add_comma(12345678)
    comma_added_12 = add_comma(12)
    
    print(comma_added_1234)
    print(comma_added_12345678)
    print(comma_added_12)
    
main()

1,234
12,345,678

def tokenize(trg, N=1):
    text = trg.split()
    #print("text",text)
    result = []
    test = ""
    
    if N == 1:
        for i in range(len(text)):
            test = str(text[i])
            result.append(test)   
            
    for j in range(len(text)-1):
        for k in range(1, N):
            test = str(text[j]) + " " + str(text[j+k])
            result.append(test)
        
    #print("rslt", result)           
    return result
        
def main():
    a = "There was a farmer who had a dog ."
    print(tokenize(a))
    print(tokenize(a, 2))
    
main()

['There', 'was', 'a', 'farmer', 'who', 'had', 'a', 'dog', '.']

def mean_and_var(*val):
    m_x, m_y = 0.0, 0.0
    var_x, var_y = 0.0, 0.0 
    
    for i in val:
        m_x += (i[0] / len(val))
        m_y += (i[1] / len(val))
        
    for j in val:    
        var_x += ((j[0] - m_x) ** 2) / len(val)
        var_y += ((j[1] - m_y) ** 2) / len(val)
    return ((m_x, m_y), (var_x, var_y))

def main():
    v1 = (0, 1)
    v2 = (0.5, 0.5)
    v3 = (1, 0)
    m, var = mean_and_var(v1, v2, v3)
    
    print(' : ', m, '\n : ', var)
    
main()
