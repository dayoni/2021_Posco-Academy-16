  • #### 소수(Prime Number) 판단하는 함수 정의하기
    • ##### 소수이면, True 반환
    • ###### 소수가 아니면, False 반환

def check_prime(num):
    for i in range(2, num):
        if num % i == 0:
            return False
        else:
            continue
    return True
    
def main():
    a = 13
    b = 15

    if check_prime(a):
        print(str(a) + '는 소수입니다.')
    else:
        print(str(a) + '는 소수가 아닙니다.')
        
    if check_prime(b):
        print(str(b) + '는 소수입니다.')
    else:
        print(str(b) + '는 소수가 아닙니다.')

main()

13는 소수입니다.
15는 소수가 아닙니다.

문제 2¶

  • #### 천 단위마다 쉼표(,) 추가하는 함수 정의하기
  • #### 1000000(int)이면, ‘1,000,000’(str) 반환
In [2]:
def add_comma(val):
    result, comma = "", ""
    cnt = 0

    number = str(val)
    number = number[len(number)::-1]

    for i in range(len(number)):
        cnt += 1
        if cnt % 3 == 0:
            result += number[i] + ','
        else :
            result += number[i]

    return result[::-1] + comma

def main():
    comma_added_1234 = add_comma(1234)
    comma_added_12345678 = add_comma(12345678)
    comma_added_12 = add_comma(12)
    
    print(comma_added_1234)
    print(comma_added_12345678)
    print(comma_added_12)
    
main()

1,234
12,345,678
12

def tokenize(trg, N=1):
    text = trg.split()
    #print("text",text)
    result = []
    test = ""
    
    if N == 1:
        for i in range(len(text)):
            test = str(text[i])
            result.append(test)   
            
    for j in range(len(text)-1):
        for k in range(1, N):
            test = str(text[j]) + " " + str(text[j+k])
            result.append(test)
        
    #print("rslt", result)           
    return result
        
def main():
    a = "There was a farmer who had a dog ."
    print(tokenize(a))
    print(tokenize(a, 2))
    
main()

['There', 'was', 'a', 'farmer', 'who', 'had', 'a', 'dog', '.']
['There was', 'was a', 'a farmer', 'farmer who', 'who had', 'had a', 'a dog', 'dog .']

문제 4¶

  • #### 여러 벡터를 전달받아, 평균(Mean)과 분산 (Variance)을 구하는 함수 정의하기

def mean_and_var(*val):
    m_x, m_y = 0.0, 0.0
    var_x, var_y = 0.0, 0.0 
    
    for i in val:
        m_x += (i[0] / len(val))
        m_y += (i[1] / len(val))
        
    for j in val:    
        var_x += ((j[0] - m_x) ** 2) / len(val)
        var_y += ((j[1] - m_y) ** 2) / len(val)
    return ((m_x, m_y), (var_x, var_y))

def main():
    v1 = (0, 1)
    v2 = (0.5, 0.5)
    v3 = (1, 0)
    m, var = mean_and_var(v1, v2, v3)
    
    print('평균 : ', m, '\n분산 : ', var)
    
main()

평균 :  (0.5, 0.5) 
분산 :  (0.16666666666666666, 0.16666666666666666)
